<IfModule mod_rewrite.c>
  RewriteEngine On
  RewriteBase /
  
  # Handle Next.js static files
  RewriteCond %{REQUEST_URI} ^/_next/static/
  RewriteRule ^(_next/static/.*) $1 [L]
  
  # Handle Next.js data files
  RewriteCond %{REQUEST_URI} ^/_next/data/
  RewriteRule ^(_next/data/.*) $1 [L]
  
  # Handle Next.js image optimization
  RewriteCond %{REQUEST_URI} ^/_next/image/
  RewriteRule ^(_next/image/.*) $1 [L]
  
  # Send all other requests to index.html
  RewriteCond %{REQUEST_FILENAME} !-f
  RewriteCond %{REQUEST_FILENAME} !-d
  RewriteRule ^(.*)$ index.html [QSA,L]
</IfModule>

# Set correct MIME types
<IfModule mod_mime.c>
  AddType application/javascript .js
  AddType application/json .json
  AddType text/css .css
  AddType image/svg+xml .svg
</IfModule>

# Enable CORS
<IfModule mod_headers.c>
  Header set Access-Control-Allow-Origin "*"
</IfModule>

# Disable directory listing
Options -Indexes

# Set caching for static assets
<IfModule mod_expires.c>
  ExpiresActive On
  
  # Images
  ExpiresByType image/jpeg "access plus 1 year"
  ExpiresByType image/png "access plus 1 year"
  ExpiresByType image/svg+xml "access plus 1 year"
  
  # CSS and JavaScript
  ExpiresByType text/css "access plus 1 month"
  ExpiresByType application/javascript "access plus 1 month"
  
  # Fonts
  ExpiresByType application/font-woff "access plus 1 year"
  ExpiresByType application/font-woff2 "access plus 1 year"
</IfModule>
